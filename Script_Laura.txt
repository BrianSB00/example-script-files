SET SERVEROUTPUT ON SIZE UNLIMITED;

DECLARE
    CURSOR c_table_name IS
        SELECT 
            NVL(column1, 0) AS column1,
            NVL(column2, ' ') AS column2,
            TO_CHAR(date_column, 'YYYY-MM-DD HH24:MI:SS') AS date_column
        FROM TABLE_NAME
        WHERE [EXACT_WHERE_CONDITIONS]
        AND ROWNUM <= 10;
BEGIN
    DBMS_OUTPUT.PUT_LINE('[COLUMN_NAMES_SEPARATED_BY_^]');
    
    FOR l_record in c_table_name LOOP 
        DBMS_OUTPUT.PUT_LINE(
            l_record.column1 || '^' ||
            l_record.column2 || '^' ||
            NVL(TO_CHAR(l_record.date_column, 'YYYY-MM-DD HH24:MI:SS'), ' ') || '^'
        );
    END LOOP;
 
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No records found for the specified criteria.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
END;
/
