SET SERVEROUTPUT ON SIZE UNLIMITED;

DECLARE
    CURSOR c_table_name IS
        SELECT 
            [ALL_COLUMNS_WITH_PROPER_FORMATTING]
        FROM TABLE_NAME
        WHERE [EXACT_WHERE_CONDITIONS]
        AND ROWNUM <= 10;
BEGIN
    DBMS_OUTPUT.PUT_LINE('[COLUMN_NAMES_SEPARATED_BY_^]');
    
    FOR l_record in c_table_name LOOP 
        DBMS_OUTPUT.PUT_LINE(
            [FORMATTED_COLUMN_VALUES_SEPARATED_BY_^]
        );
    END LOOP;
 
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No records found for the specified criteria.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
END;
/
